MySQL Stored Procedures - Full Tutorial

What is a Stored Procedure?

A Stored Procedure is a saved SQL code that you can run anytime using CALL. It helps in reusing SQL queries and improving performance.

Basic Syntax

DELIMITER $$

CREATE PROCEDURE procedure_name()
BEGIN
    SQL statements;
END $$

DELIMITER ;

Create a Simple Procedure

DELIMITER $$

CREATE PROCEDURE showStudents()
BEGIN
    SELECT * FROM student;
END $$

DELIMITER ;

Call the Procedure

CALL showStudents();

Procedure with IN Parameter

DELIMITER $$

CREATE PROCEDURE getStudentById(IN sid INT)
BEGIN
    SELECT * FROM student WHERE id = sid;
END $$

DELIMITER ;
    
CALL getStudentById(5);

Procedure with OUT Parameter

DELIMITER $$

CREATE PROCEDURE countStudents(OUT total INT)
BEGIN
    SELECT COUNT(*) INTO total FROM student;
END $$

DELIMITER ;
    
CALL countStudents(@t);
SELECT @t;

Procedure with Multiple Parameters

DELIMITER $$

CREATE PROCEDURE addStudent(IN n VARCHAR(50), IN m INT)
BEGIN
    INSERT INTO student(name, marks) VALUES(n, m);
END $$

DELIMITER ;
    
CALL addStudent('Bhavesh', 88);

Procedure with IF / ELSE

DELIMITER $$

CREATE PROCEDURE checkMarks(IN m INT)
BEGIN
    IF m >= 40 THEN
        SELECT 'Pass';
    ELSE
        SELECT 'Fail';
    END IF;
END $$

DELIMITER ;
    
CALL checkMarks(35);

Procedure with LOOP

DELIMITER $$

CREATE PROCEDURE printNumbers()
BEGIN
    DECLARE x INT DEFAULT 1;

    WHILE x <= 5 DO
        SELECT x;
        SET x = x + 1;
    END WHILE;
END $$

DELIMITER ;

Update Using Stored Procedure

DELIMITER $$

CREATE PROCEDURE updateMarks(IN sid INT, IN newMarks INT)
BEGIN
    UPDATE student 
    SET marks = newMarks 
    WHERE id = sid;
END $$

DELIMITER ;
    
CALL updateMarks(3, 90);

Delete Stored Procedure

DROP PROCEDURE showStudents;
